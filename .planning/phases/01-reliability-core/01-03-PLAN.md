### Task 1: Unit tests for retry-util

**Files:** `print-agent-desktop/agent/retry-util.test.js`

**Action:**
Criar testes Jest/Mocha para `retry-util.js`:

**Test cases:**
1. **Success on first attempt** - Função sucede sem retries
2. **Success after retries** - Função falha 2x, sucede na 3ª
3. **Failure after max retries** - Função falha 3x, throw error final
4. **Exponential backoff timing** - Verificar delays: 1s, 2s, 4s
5. **Custom retry config** - maxRetries=2, baseDelay=500ms

**Implementação:**
```javascript
const { retryWithBackoff } = require('./retry-util');

describe('retryWithBackoff', () => {
  it('succeeds on first attempt', async () => {
    const fn = jest.fn().mockResolvedValue('success');
    const result = await retryWithBackoff(fn, 3, 100);
    expect(result).toBe('success');
    expect(fn).toHaveBeenCalledTimes(1);
  });

  it('retries on failure and succeeds', async () => {
    const fn = jest.fn()
      .mockRejectedValueOnce(new Error('fail'))
      .mockRejectedValueOnce(new Error('fail'))
      .mockResolvedValue('success');
    
    const result = await retryWithBackoff(fn, 3, 100);
    expect(result).toBe('success');
    expect(fn).toHaveBeenCalledTimes(3);
  });

  it('throws after max retries', async () => {
    const fn = jest.fn().mockRejectedValue(new Error('fail'));
    await expect(retryWithBackoff(fn, 3, 100)).rejects.toThrow('fail');
    expect(fn).toHaveBeenCalledTimes(3);
  });

  it('uses exponential backoff', async () => {
    const fn = jest.fn().mockRejectedValue(new Error('fail'));
    const start = Date.now();
    
    try {
      await retryWithBackoff(fn, 3, 100);
    } catch (e) {}
    
    const elapsed = Date.now() - start;
    // 100ms + 200ms + 400ms = 700ms (com margem)
    expect(elapsed).toBeGreaterThanOrEqual(600);
    expect(elapsed).toBeLessThan(900);
  });
});
```

### Task 2: Unit tests for circuit-breaker

**Files:** `print-agent-desktop/agent/circuit-breaker.test.js`

**Action:**
Criar testes para `circuit-breaker.js`:

**Test cases:**
1. **Initial state is CLOSED** - Estado inicial correto
2. **Opens after threshold failures** - 5 falhas → OPEN
3. **Blocks requests when OPEN** - Requests lançam erro quando OPEN
4. **Transitions to HALF_OPEN after timeout** - Após 30s, permite 1 request
5. **Closes on success in HALF_OPEN** - Success → CLOSED
6. **Reopens on failure in HALF_OPEN** - Failure → OPEN novamente
7. **Resets failure count on success** - Success zera contador

**Implementação:**
```javascript
const { CircuitBreaker } = require('./circuit-breaker');

describe('CircuitBreaker', () => {
  it('starts in CLOSED state', () => {
    const cb = new CircuitBreaker(3, 1000);
    expect(cb.getState().state).toBe('CLOSED');
  });

  it('opens after threshold failures', async () => {
    const cb = new CircuitBreaker(3, 1000);
    const fn = jest.fn().mockRejectedValue(new Error('fail'));
    
    for (let i = 0; i < 3; i++) {
      try { await cb.execute(fn); } catch (e) {}
    }
    
    expect(cb.getState().state).toBe('OPEN');
  });

  it('blocks requests when OPEN', async () => {
    const cb = new CircuitBreaker(2, 1000);
    const fn = jest.fn().mockRejectedValue(new Error('fail'));
    
    try { await cb.execute(fn); } catch (e) {}
    try { await cb.execute(fn); } catch (e) {}
    
    await expect(cb.execute(fn)).rejects.toThrow('Circuit breaker is OPEN');
  });

  it('transitions to HALF_OPEN after timeout', async () => {
    jest.useFakeTimers();
    const cb = new CircuitBreaker(2, 1000);
    const fn = jest.fn().mockRejectedValue(new Error('fail'));
    
    try { await cb.execute(fn); } catch (e) {}
    try { await cb.execute(fn); } catch (e) {}
    
    expect(cb.getState().state).toBe('OPEN');
    
    jest.advanceTimersByTime(1001);
    
    fn.mockResolvedValue('success');
    await cb.execute(fn);
    
    expect(cb.getState().state).toBe('CLOSED');
    jest.useRealTimers();
  });
});
```

### Task 3: Integration tests for reliability features

**Files:** `print-agent-desktop/agent/reliability.integration.test.js`

**Action:**
Criar testes de integração end-to-end:

**Test cases:**
1. **Print job retries on temporary failure** - Mock impressora offline, verifica retry
2. **Print job fails after max retries** - Mock falha permanente, verifica desistência
3. **Circuit breaker pauses polling on API offline** - Mock API offline, verifica pausa
4. **Circuit breaker resumes on API recovery** - Mock API volta, verifica retomada
5. **Retry + Circuit breaker interaction** - Ambos funcionam juntos

**Implementação:**
```javascript
const { printJob } = require('./printer');
const { CircuitBreaker } = require('./circuit-breaker');
const apiClient = require('./api-client');

describe('Reliability Integration', () => {
  it('retries print job on temporary failure', async () => {
    const mockPrint = jest.spyOn(require('./pdf-printer'), 'printPDF')
      .mockRejectedValueOnce(new Error('Printer offline'))
      .mockRejectedValueOnce(new Error('Printer offline'))
      .mockResolvedValue(true);
    
    const job = { content: { type: 'pdf', content: 'base64...' } };
    const result = await printJob(job);
    
    expect(result).toBe(true);
    expect(mockPrint).toHaveBeenCalledTimes(3);
  });

  it('circuit breaker pauses polling on API offline', async () => {
    const cb = new CircuitBreaker(3, 1000);
    const mockAPI = jest.spyOn(apiClient, 'getPendingJobs')
      .mockRejectedValue(new Error('ECONNREFUSED'));
    
    for (let i = 0; i < 3; i++) {
      try { await cb.execute(() => apiClient.getPendingJobs('uuid')); } 
      catch (e) {}
    }
    
    expect(cb.getState().state).toBe('OPEN');
    
    await expect(
      cb.execute(() => apiClient.getPendingJobs('uuid'))
    ).rejects.toThrow('Circuit breaker is OPEN');
  });
});
```
